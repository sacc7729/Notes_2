# Basics

```
// print
print("hello")
// list
[1, 2, 3]

// control
for car in cars:
  if car == "toyota":
    print("cool")

// dict
my_dict["my name"] = "my address"

// function
def add_three(x):
  return x + 3

// class
class Dog():
  def __init__(self, name):
    self.name = name
  def sit():
    print(self.name + " sat")

// inheritance
class SmallDog(Dog):
  def __init__(self, name):
    super().__init__(name)
    self.size = "small"
```
